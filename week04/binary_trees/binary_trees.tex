\section{Tré}

\begin{frame}{Hvað eru tré}
    \begin{itemize}
        \item Tré koma við sögu mjög víða innan tölvunarfræðinnar
        \item Tré eru notuð til að tákna tengsl
        \item Nánar tiltekið stigveldi (e. hierarchy)
    \end{itemize}
\end{frame}

\begin{frame}{Orðaforði}
    \begin{itemize}
        \item Foreldri (parent) og börn (children)
        \item Systkini (siblings)
        \item Rót (root)
        \item Lauf (leaf) og innri hnútar (internal node) (ytri hnútar (external nodes))
        \item Stígur (path)
        \item Forfaðir (ancestor) og afkomandi (desecendant)
    \end{itemize}
\end{frame}

\begin{frame}{Orðaforði}
    \begin{itemize}
        \item Dýpt (deapth) hnútar $u$ er lengd stígsins frá rótinni til $u$
        \item Stig (level) $n$ er mengi allra hnúta af dýpt $n$
        \item Hæð (height) trés er hæsta dýpt hnútar í trénu
        \item Hluttré (subtree)
    \end{itemize}
\end{frame}

\begin{frame}{Tvíundartré}
    \begin{itemize}
        \item Við munum aðallega skoða tvíundartré (e. binary trees)
        \item Tvíundartré eru tré þar sem hver hnútur á í mesta lagi tvö börn
    \end{itemize}
\end{frame}


\section{Skilgreiningar}

\begin{frame}{Tengdir listar}
    Hægt er að skilgreina (tengda) lista endurkvæmt
    \begin{block}{Heiltölulisti}
        \begin{itemize}
            \item $\nil$ er heiltölulisti
            \item Ef $n$ er heiltala og $\ell$ er heiltölulisti, þá er $n::\ell$ heiltölulisti
            \item Ekkert annað er heiltölulisti
        \end{itemize}
    \end{block}
    Hvað græðum við á því?
\end{frame}

\begin{frame}{Tvíundartré}
    Hægt er að skilgreina tvíundartré endurkvæmt
    \begin{block}{Heiltölutvíundartré}
        \begin{itemize}
            \item $\Lambda$ er heiltölutvíundartré
            \item Ef $n$ er heiltala og $T_1$ og $T_2$ eru heiltölutvíundartré, þá er $\langle n, T_1, T_2 \rangle$ heiltölutvíundartré
            \item Ekkert annað er heiltölutvíundartré
        \end{itemize}
    \end{block}
    Hvað græðum við á því?
\end{frame}
