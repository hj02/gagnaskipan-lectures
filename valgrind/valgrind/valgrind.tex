\begin{frame}{Algengustu villur}
    \begin{itemize}
        \item Invalid read/write
        \item Conditional jump or move depends on uninitialised value(s)
        \item Invalid free()
    \end{itemize}
\end{frame}

\begin{frame}{Invalid read/write}
    \begin{itemize}
        \item Forritið les úr (skrifa í) minni sem það hefur ekki aðgang að
        \item Þrennt kemur til greina
            \begin{itemize}
                \item Forritið hefur lesið (skrifað) út fyrir löglegt minni (t.d. vísað út fyrir fylki)
                \item Forritið hefur lesið úr (skrifað í) \cpp|NULL|
                \item Forritið hefur lesið úr (skrifað í) minni sem búið var að skila með \cpp|delete|
            \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Conditional jump or move depends on uninitialised value(s)}
    \begin{itemize}
        \item Forritið notar ófrumstillta breytu inni í skilyrði í if-setningu eða lykkju
    \end{itemize}
\end{frame}

\begin{frame}{Invalid free()}
    \begin{itemize}
        \item Forritið skilaði minni (með \cpp|delete|) sem það hafði ekki aðgang að
        \item Langalgengasta orsök: verið er að skila sama minni tvisvar
    \end{itemize}
\end{frame}

\begin{frame}{Sjaldgæfari villur}
    \begin{itemize}
        \item Mismatched free() / delete / delete []
        \item Argument `size' of function malloc has a fishy (possibly negative) value
            \begin{itemize}
                \item Mooshak er með eldri útgáfu af Valgrind
                \item Warning: silly arg (-4) to \_\_builtin\_vec\_new()
            \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Mismatched free() / delete / delete []}
    \begin{itemize}
        \item Forritið kallar á \cpp|delete []| þegar það ætti að kalla á \cpp|delete| eða öfugt
        \item T.d. þegar fylki er skilað með \cpp|delete| en ekki \cpp|delete []|
    \end{itemize}
\end{frame}

\begin{frame}{Argument `size' of function malloc has a fishy value}
    \begin{itemize}
        \item Forritið kallar á \cpp|new| með einkennilegri (oftast neikværði) tölu
    \end{itemize}
\end{frame}

\begin{frame}{Minnisleki}
    \begin{itemize}
        \item ?? bytes in ?? blocks are definately lost
        \item Forritið tekur frá minni (með \cpp|new|) sem það skilar ekki (með \cpp|delete|)
            \begin{itemize}
                \item Valgrind sýnir aðeins hvar minnið var tekið frá
                \item Valgrind veit ekki hvar minninu átti að vera skilað!
            \end{itemize}
    \end{itemize}
\end{frame}
