\section{Staflar}

\begin{frame}{Staflar}
    \begin{itemize}
        \item Stafli (e. stack) er hugrænt gangatag sem geymir safn hluta og leyfir tvær aðgerir
            \begin{itemize}
                \item Push: Bætir staki "`efst"' á staflann
                \item Pop: Fjarlægir "`efsta"' stakið á staflann
            \end{itemize}
        \item LIFO (Last In First Out)
    \end{itemize}
\end{frame}

\begin{frame}[fragile]
    \frametitle{Skil}
    \begin{lstlisting}[title={Heiltölustafli}]
class IntStack() {
    // Initializes an empty stack.
    IntStack();
    // Determines whether the stack is empty.
    bool empty();
    // Adds 'item' to the top of the stack.
    void push(int item);
    // Removes the top of the stack (i.e. removes the element that was added
    // most recently), and returns it.
    int pop();
    // Retrieves the top of the stack.
    int top();
};
    \end{lstlisting}
\end{frame}

\section{Biðraðir}

\begin{frame}{Biðraðir}
    \begin{itemize}
        \item Biðröð (e. queue) er hugrænt gangatag sem geymir safn hluta og leyfir tvær aðgerir
            \begin{itemize}
                \item Enqueue: Bætir staki "`fremst"' í biðröðina
                \item Dequeue: Fjarlægir "`síðasta"' stakið úr biðröðinni
            \end{itemize}
        \item FIFO (First In First Out)
    \end{itemize}
\end{frame}

\begin{frame}[fragile]
    \frametitle{Skil}
    \begin{lstlisting}[title={Heiltölubiðröð}]
class IntQueue() {
    // Initializes an empty queue.
    IntQueue();
    // Determines whether the queue is empty.
    bool empty();
    // Inserts 'item' at the back of the queue.
    void enqueue(int item);
    // Removes and returns the front of the queue (i.e. the element that
    // was enqueued earliest).
    int dequeue();
    // Returns the front of the queue.
    int front();
}
    \end{lstlisting}
\end{frame}

\begin{frame}{Reverse polish Notation}
    \begin{itemize}
        \item Reverse Polish Notation (postfix notation)
        \item Virkjar (e. operators) koma á eftir þolendum (e. opearand)
    \end{itemize}
    \begin{block}{RPN}
        Eftirfarandi mállýsing (e. grammar) lýsir RPN
        \begin{align*}
            \textrm{Expr} &\Rightarrow \textrm{Num} \\
            \textrm{Expr} &\Rightarrow \textrm{Expr} \ \ \textrm{Expr} \ \ \textrm{Op} \\
            \textrm{Op} &\Rightarrow + \ | \ - \ | \ *
        \end{align*}
    \end{block}
\end{frame}
